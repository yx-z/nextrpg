nextrpg.scene.map.loader

Map helper system for nextrpg.

This module provides helper classes and functions for loading and processing TMX tile maps in nextrpg games. It includes utilities for tile grouping, polygon creation, collision detection, and map rendering.

Features:
  • Tile grouping and bottom pixel calculation
  • Polygon and rectangle creation from Tiled objects
  • Foreground/background/above-character layer management
  • Collision detection and debug visualization
  • Map object and layer access utilities

 46class TileBottomAndDrawOnScreen(NamedTuple):
 47    """
 48    Represents a foreground tile of the bottommost pixel from its tile class,
 49    and a `DrawOnScreen` of the tile itself.
 50
 51    This is particularly useful for foreground tiles when performing y-sorting
 52    to determine the laying between the character and the tile: bigger y-axis
 53    (closer to the bottom) means closer to the camera and hence obstructing
 54    tiles/characters above it.
 55
 56    We need to group the tiles given a single graphical object may be split
 57    into multiple tiles in a Tiled tmx map.
 58
 59    Attributes:
 60        `bottom`: The bottommost pixel for the tile group that this tile is in.
 61
 62        `draw`: The `DrawOnScreen` of the tile itself.
 63    """
 64
 65    bottom: Pixel
 66    draw: DrawOnScreen

 86def get_polygon(obj: TiledObject) -> PolygonOnScreen | None:
 87    """
 88    Create a polygon from a Tiled object on a map.
 89
 90    Arguments:
 91        `obj`: The Tiled object to create a polygon from.
 92
 93    Returns:
 94        `PolygonOnScreen`: The polygon created from the Tiled object.
 95    """
 96    if hasattr(obj, "points"):
 97        return PolygonOnScreen(
 98            tuple(Coordinate(x, y) for x, y in obj.points), obj.closed
 99        )
100    if _is_rect(obj):
101        return RectangleOnScreen(
102            Coordinate(obj.x, obj.y), Size(obj.width, obj.height)
103        )
104    return None

299    def _from_points(
300        self, coord: _TileCoordinate, obj: TiledObject
301    ) -> PolygonOnScreen:
302        w, h = self._tile_size
303        cx, cy = coord
304        return PolygonOnScreen(

308    def _from_rect(
309        self, coord: _TileCoordinate, obj: TiledObject
310    ) -> RectangleOnScreen | None:
311        if not _is_rect(obj):
312            return None
313
314        w, h = self._tile_size
315        cx, cy = coord
316        map_coord = Coordinate(cx * w + obj.x, cy * h + obj.y)
317        size = Size(obj.width, obj.height)
318        return RectangleOnScreen(map_coord, size)

338    def _bottom_and_draw(
339        self, layer: TiledTileLayer
340    ) -> tuple[TileBottomAndDrawOnScreen, ...]:
341        coord_and_draws = self._draw(layer)
342        coord_to_bottom = {
343            coord: draw.visible_rectangle_on_screen.bottom
344            for coord, draw in coord_and_draws.items()
345        }
346        bottom_and_draw = tuple(
347            TileBottomAndDrawOnScreen(
348                self._bottom(layer, coord, draw, coord_to_bottom), draw
349            )
350            for coord, draw in coord_and_draws.items()
351        )
352        return tuple(sorted(bottom_and_draw, key=lambda t: t.bottom))

354    @property
355    def _tile_size(self) -> Size:
356        return Size(self._tmx.tilewidth, self._tmx.tileheight)

358    def _draw(
359        self, layer: TiledTileLayer
360    ) -> dict[_TileCoordinate, DrawOnScreen]:
361        return {
362            _TileCoordinate(top, left): self._tile(left, top, surface)
363            for left, top, surface in layer.tiles()
364        }
365
366    def _tile(self, left: int, top: int, surface: Surface) -> DrawOnScreen:
367        width, height = self._tile_size
368        return DrawOnScreen(
369            Coordinate(left * width, top * height),
370            Draw(surface),
371        )

373    def _class(
374        self, layer: TiledTileLayer, coord: _TileCoordinate
375    ) -> str | None:
376        x, y = coord
377        if 0 <= x < len(layer.data) and 0 <= y < len(layer.data[x]):
378            data_id = layer.data[x][y]
379            return self._gid_to_cls.get(
380                self._tmx.tile_properties.get(data_id, {}).get("id")
381            )
382        return None
383
384    def _bottom(
385        self,
386        layer: TiledTileLayer,
387        coord: _TileCoordinate,
388        draw: DrawOnScreen,
389        coord_to_bottom: dict[_TileCoordinate, Pixel],
390    ) -> Pixel:
391        return (
392            max(coord_to_bottom[c] for c in self._component(layer, coord, cls))
393            if (cls := self._class(layer, coord))
394            else draw.visible_rectangle_on_screen.bottom
395        )
396
397    def _component(
398        self,
399        layer: TiledTileLayer,
400        coord: _TileCoordinate,
401        cls: str,
402        visited: set[_TileCoordinate] | None = None,
403    ) -> set[_TileCoordinate]:
404        visited = (visited or set()) | {coord}
405        return visited | {
406            c
407            for neighbor in self._neighbors(layer, coord, cls) - visited
408            for c in self._component(layer, neighbor, cls, visited)
409        }
410
411    def _neighbors(
412        self, layer: TiledTileLayer, coord: _TileCoordinate, cls: str
413    ) -> set[_TileCoordinate]:
414        x, y = coord
415        return {
416            c
417            for dx, dy in ((0, 1), (1, 0), (0, -1), (-1, 0))
418            if self._class(layer, c := _TileCoordinate(x + dx, y + dy)) == cls
419        }
420
421    @cached_property
422    def _gid_to_cls(self) -> dict[_Gid, str]:
423        return {
424            tile["id"]: cls
425            for tile in self._tmx.tile_properties.values()
426            if (cls := tile.get("type"))
427        }

438def _below_character_layer(
439    layer: tuple[LayerTileBottomAndDrawOnScreen, ...],
440    character: CharacterOnScreen,
441) -> bool:
442    rect = character.draw_on_screen.visible_rectangle_on_screen
443    return layer[-1].bottom < rect.bottom and any(
444        bottom < rect.bottom and rect.collide(draw.visible_rectangle_on_screen)
445        for _, bottom, draw in layer
446    )

374    def _class(
375        self, layer: TiledTileLayer, coord: _TileCoordinate
376    ) -> str | None:
377        x, y = coord
378        if 0 <= x < len(layer.data) and 0 <= y < len(layer.data[x]):
379            data_id = layer.data[x][y]
380            return self._gid_to_cls.get(

386        self,
387        layer: TiledTileLayer,
388        coord: _TileCoordinate,
389        draw: DrawOnScreen,
390        coord_to_bottom: dict[_TileCoordinate, Pixel],
391    ) -> Pixel:
392        return (
393            max(coord_to_bottom[c] for c in self._component(layer, coord, cls))
394            if (cls := self._class(layer, coord))
395            else draw.visible_rectangle_on_screen.bottom
396        )

398    def _component(
399        self,
400        layer: TiledTileLayer,
401        coord: _TileCoordinate,
402        cls: str,
403        visited: set[_TileCoordinate] | None = None,
404    ) -> set[_TileCoordinate]:
405        visited = (visited or set()) | {coord}
406        return visited | {
407            c
408            for neighbor in self._neighbors(layer, coord, cls) - visited
409            for c in self._component(layer, neighbor, cls, visited)
410        }

413        self, layer: TiledTileLayer, coord: _TileCoordinate, cls: str
414    ) -> set[_TileCoordinate]:
415        x, y = coord
416        return {
417            c
418            for dx, dy in ((0, 1), (1, 0), (0, -1), (-1, 0))
419            if self._class(layer, c := _TileCoordinate(x + dx, y + dy)) == cls
420        }
