nextrpg.draw.draw

Drawable elements and rendering utilities for nextrpg.

This module provides the core drawing system for nextrpg games. It defines classes for representing drawable elements, managing their positioning on screen, and handling various drawing operations.

Features:
  • Drawing: Represents a drawable element with size and surface access
  • DrawOnScreen: Combines a drawing with its position on screen
  • Polygon: Represents polygon shapes with collision detection
  • Rectangle: Specialized rectangle with optimized collision detection

 41class Draw:
 42    """
 43    Represents a drawable element and provides methods for accessing its size
 44    and dimensions.
 45
 46    This class loads a surface from a file or directly accepts a
 47    `pygame.Surface` as input. It provides properties to access surface
 48    dimensions and size and methods to crop and scale the surface.
 49
 50    Arguments:
 51        `resource`: A path to a file containing the drawing resource,
 52            or a `pygame.Surface` object.
 53
 54    Example:
 55        ```python
 56        # Load from file
 57        sprite = Drawing("assets/player.png")
 58
 59        # Use existing surface
 60        surface = pygame.Surface((32, 32))
 61        drawing = Drawing(surface)
 62
 63        # Get dimensions
 64        size = drawing.size  # Returns Size(width, height)
 65        width = drawing.width  # Returns int
 66        ```
 67    """
 68
 69    resource: str | PathLike | Surface
 70
 71    @property
 72    def width(self) -> Pixel:
 73        """
 74        Gets the width of the surface.
 75
 76        Returns:
 77            `Pixel`: The width of the surface in pixel measurement.
 78        """
 79        return self._surface.width
 80
 81    @property
 82    def height(self) -> Pixel:
 83        """
 84        Gets the height of the surface.
 85
 86        Returns:
 87            `Pixel`: The height of the surface in pixel measurement.
 88        """
 89        return self._surface.height

 91    @property
 92    def size(self) -> Size:
 93        """
 94        Gets the size of an object as a combination of its width and height.
 95
 96        Returns:
 97            `Size`: A Size object containing the width and height of the object.
 98        """
 99        return Size(self.width, self.height)

101    @property
102    def pygame(self) -> Surface:
103        """
104        Gets the current `pygame.Surface` for the object.
105
106        Returns:
107            `pygame.Surface`: The underlying `pygame.Surface`.
108        """
109        return self._debug_surface or self._surface

111    def crop(self, top_left: Coordinate, size: Size) -> Self:
112        """
113        Crops a rectangular portion of the drawing specified by the
114        top-left corner and the size.
115
116        The method extracts a subsection of the drawing based on the provided
117        coordinates and dimensions and returns a new `Drawing` instance.
118        The original drawing remains unchanged.
119
120        Arguments:
121            `top_left`: The top-left coordinate of the rectangle to be cropped.
122
123            `size`: The width and height of the rectangle to be cropped.
124
125        Returns:
126            `Draw`: A new `Drawing` instance representing the cropped area.
127
128        Example:
129            ```python
130            drawing = Drawing("spritesheet.png")

132            sprite = drawing.crop(Coordinate(64, 0), Size(32, 32))
133            ```
134        """
135        left, top = top_left
136        width, height = size

307@dataclass(frozen=True)
308class PolygonOnScreen:
309    """
310    Represents a polygon shape defined by a sequence of points.
311
312    This class provides polygon functionality including collision detection,
313    drawing operations, and geometric calculations. Polygons can be either
314    closed (default) or open shapes.
315    """
316
317    points: tuple[Coordinate, ...]
318    closed: bool = True
319
320    @cached_property
321    def length(self) -> Pixel:
322        length = sum(
323            p.distance(np) for p, np in zip(self.points, self.points[1:])
324        )
325        if self.closed:
326            return length + self.points[0].distance(self.points[-1])
327        return length
328
329    @cached_property
330    def bounding_rectangle(self) -> RectangleOnScreen:
331        return _bounding_rectangle(self.points)
332
333    @cached_property
334    def _mask(self) -> Mask:
335        return from_surface(self.fill(BLACK).drawing.pygame)
336
337    def fill(self, color: Rgba) -> DrawOnScreen:
338        """
339        Creates a filled polygon drawing with the provided color.
340
341        Arguments:
342            `color`: The color to fill the polygon with.
343
344        Returns:
345            `DrawOnScreen`: A drawing of the filled polygon.
346
347        Example:
348            ```python
349            triangle = Polygon((Coordinate(0, 0), Coordinate(50, 0),
350                              Coordinate(25, 50)))
351            filled = triangle.fill(Rgba(255, 0, 0, 255))  # Red triangle
352            ```
353        """
354        return self._draw(_fill_polygon(color))
355
356    def line(
357        self, color: Rgba, stroke_width: Pixel | None = None
358    ) -> DrawOnScreen:
359        """
360        Creates a line drawing of the polygon with the provided color.
361
362        Arguments:
363            `color`: The color of the line.
364
365            `stroke_width`: The width of the line in pixels.
366                If `None`, uses the default stroke width from global_config.
367
368        Returns:
369            `DrawOnScreen`: A drawing of the polygon outline.
370
371        Example:
372            ```python
373            triangle = Polygon((Coordinate(0, 0), Coordinate(50, 0),
374                              Coordinate(25, 50)))
375            outline = triangle.line(Rgba(0, 0, 255, 255), stroke_width=2)
376            ```
377        """
378        if stroke_width is None:
379            stroke = config().draw_on_screen.stroke_width
380        else:
381            stroke = stroke_width
382
383        def _line(surface: Surface, points: tuple[Coordinate, ...]) -> None:
384            lines(surface, color, self.closed, points, stroke)
385
386        return self._draw(_line)
387
388    def collide(self, poly: Self) -> bool:
389        """
390        Checks if this polygon overlaps with another polygon.
391
392        Performs precise collision detection using pygame masks.
393        First checks bounding rectangles for efficiency, then
394        performs pixel-perfect collision detection.
395
396        Arguments:
397            `poly`: The polygon to check for collision with.
398
399        Returns:
400            `bool`: `True` if the polygons overlap, `False` otherwise.
401
402        Example:
403            ```python
404            poly1 = Polygon((Coordinate(0, 0), Coordinate(50, 50)))
405            poly2 = Polygon((Coordinate(25, 25), Coordinate(75, 75)))
406            is_colliding = poly1.collide(poly2)  # True
407            ```
408        """
409        if not self.bounding_rectangle.collide(poly.bounding_rectangle):
410            return False
411        offset = (
412            self.bounding_rectangle.top_left - poly.bounding_rectangle.top_left
413        )
414        return bool(self._mask.overlap(poly._mask, offset))
415
416    def __contains__(self, coordinate: Coordinate) -> bool:
417        """
418        Checks if a coordinate point lies within this polygon.
419
420        The point is considered inside if it falls within the polygon's
421        bounds, including points on the edges.
422
423        Arguments:
424            `coordinate`: The coordinate point to check.
425
426        Returns:
427            `bool`: Whether the coordinate lies within the polygon.
428
429        Example:
430            ```python
431            triangle = Polygon((Coordinate(0, 0), Coordinate(50, 0),
432                              Coordinate(25, 50)))
433            is_inside = triangle.contain(Coordinate(25, 25))  # True
434            ```
435        """
436        x, y = coordinate - self.bounding_rectangle.top_left
437        width, height = self._mask.get_size()
438        if 0 <= x < width and 0 <= y < height:
439            return bool(self._mask.get_at((x, y)))
440        return False
441
442    def _draw(
443        self, method: Callable[[Surface, tuple[Coordinate, ...]], None]
444    ) -> DrawOnScreen:
445        surf = _draw_polygon(self.points, method, self.bounding_rectangle)
446        return DrawOnScreen(self.bounding_rectangle.top_left, Draw(surf))
447
448    def __add__(self, coordinate: Coordinate) -> Self:
449        points = tuple(t + coordinate for t in self.points)
450        return replace(self, points=points)
451
452    def __sub__(self, coordinate: Coordinate) -> Self:
453        return self + -coordinate

645    @override

651            self.top_left.left < poly.top_right.left
652            and self.top_right.left > poly.top_left.left
653            and self.top_left.top < poly.bottom_right.top
654            and self.bottom_right.top > poly.top_left.top
655        )

657    @override
658    def __contains__(self, coordinate: Coordinate) -> bool:
659        return (
660            self.left < coordinate.left < self.right
661            and self.top < coordinate.top < self.bottom
662        )

695def _bounding_rectangle(points: tuple[Coordinate, ...]) -> RectangleOnScreen:
696    min_x = min(c.left for c in points)
697    min_y = min(c.top for c in points)
698    max_x = max(c.left for c in points)
699    max_y = max(c.top for c in points)
700    coord = Coordinate(min_x, min_y)
701    size = Size(max_x - min_x, max_y - min_y)
702    return RectangleOnScreen(coord, size)

705def _draw_polygon(
706    points: tuple[Coordinate, ...],
707    method: Callable[[Surface, tuple[Coordinate, ...]], None],
708    bounding_rectangle: RectangleOnScreen | None = None,
709) -> Surface:
710    bounding_rectangle = bounding_rectangle or _bounding_rectangle(points)
711    surf = Surface(bounding_rectangle.size, SRCALPHA)
712    negated = tuple(p - bounding_rectangle.top_left for p in points)
713    method(surf, negated)
714    return surf
