nextrpg.draw.draw_on_screen

Drawable elements and rendering utilities for nextrpg.

This module provides the core drawing system for nextrpg games. It defines classes for representing drawable elements, managing their positioning on screen, and handling various drawing operations.

Features:
  • Drawing: Represents a drawable element with size and surface access
  • DrawOnScreen: Combines a drawing with its position on screen
  • Polygon: Represents polygon shapes with collision detection
  • Rectangle: Specialized rectangle with optimized collision detection

 42    """
 43    Represents a drawable element and provides methods for accessing its size

 45

 60        surface = pygame.Surface((32, 32))

297class Polygon:
298    """
299    Represents a polygon shape defined by a sequence of points.
300
301    This class provides polygon functionality including collision detection,
302    drawing operations, and geometric calculations. Polygons can be either
303    closed (default) or open shapes.
304
305    Arguments:
306        `points`: A tuple of coordinates defining the polygon vertices.
307
308        `closed`: Whether the polygon is closed (last point connects to first).
309            Defaults to `True`.
310
311    Example:
312        ```python
313        # Create a triangle
314        triangle = Polygon((
315            Coordinate(0, 0),
316            Coordinate(50, 0),
317            Coordinate(25, 50)
318        ))
319
320        # Create an open line
321        line = Polygon((
322            Coordinate(0, 0),
323            Coordinate(100, 100)
324        ), closed=False)
325        ```
326    """
327
328    points: tuple[Coordinate, ...]
329    closed: bool = True
330
331    @cached_property
332    def bounding_rectangle(self) -> Rectangle:
333        """
334        Get a rectangle bounding the polygon.
335
336        Calculates the smallest rectangle that completely contains
337        all points of the polygon.
338
339        Returns:
340            `Rectangle`: A rectangle bounding the polygon.
341
342        Example:
343            ```python
344            polygon = Polygon((Coordinate(0, 0), Coordinate(50, 50)))
345            bounds = polygon.bounding_rectangle
346            ```
347        """
348        min_x = min(c.left for c in self.points)
349        min_y = min(c.top for c in self.points)
350        max_x = max(c.left for c in self.points)
351        max_y = max(c.top for c in self.points)
352        coord = Coordinate(min_x, min_y)
353        size = Size(max_x - min_x, max_y - min_y)
354        return Rectangle(coord, size)
355
356    @cached_property
357    def _mask(self) -> Mask:
358        """
359        Get the collision mask for this polygon.
360
361        Creates a pygame mask from the filled polygon for precise
362        collision detection.
363
364        Returns:
365            `Mask`: The collision mask for this polygon.
366        """
367        return from_surface(self.fill(BLACK).drawing.pygame)
368
369    def fill(self, color: Rgba) -> DrawOnScreen:
370        """
371        Creates a filled polygon drawing with the provided color.
372
373        Arguments:
374            `color`: The color to fill the polygon with.
375
376        Returns:
377            `DrawOnScreen`: A drawing of the filled polygon.
378
379        Example:
380            ```python
381            triangle = Polygon((Coordinate(0, 0), Coordinate(50, 0),
382                              Coordinate(25, 50)))
383            filled = triangle.fill(Rgba(255, 0, 0, 255))  # Red triangle
384            ```
385        """
386        poly = lambda surf, points: polygon(surf, color, points)
387        return self._draw(poly)
388
389    def line(
390        self, color: Rgba, stroke_width: Pixel | None = None
391    ) -> DrawOnScreen:
392        """
393        Creates a line drawing of the polygon with the provided color.
394
395        Arguments:
396            `color`: The color of the line.
397
398            `stroke_width`: The width of the line in pixels.
399                If `None`, uses the default stroke width from global_config.
400
401        Returns:
402            `DrawOnScreen`: A drawing of the polygon outline.
403
404        Example:
405            ```python
406            triangle = Polygon((Coordinate(0, 0), Coordinate(50, 0),
407                              Coordinate(25, 50)))
408            outline = triangle.line(Rgba(0, 0, 255, 255), stroke_width=2)
409            ```
410        """
411        if stroke_width is None:
412            stroke = config().draw_on_screen.stroke_width
413        else:
414            stroke = stroke_width
415        line = lambda surf, points: lines(
416            surf, color, self.closed, points, stroke
417        )
418        return self._draw(line)
419
420    def collide(self, poly: Self) -> bool:
421        """
422        Checks if this polygon overlaps with another polygon.
423
424        Performs precise collision detection using pygame masks.
425        First checks bounding rectangles for efficiency, then
426        performs pixel-perfect collision detection.
427
428        Arguments:
429            `poly`: The polygon to check for collision with.
430
431        Returns:
432            `bool`: `True` if the polygons overlap, `False` otherwise.
433
434        Example:
435            ```python
436            poly1 = Polygon((Coordinate(0, 0), Coordinate(50, 50)))
437            poly2 = Polygon((Coordinate(25, 25), Coordinate(75, 75)))
438            is_colliding = poly1.collide(poly2)  # True
439            ```
440        """
441        if not self.bounding_rectangle.collide(poly.bounding_rectangle):
442            return False
443        offset = (
444            self.bounding_rectangle.top_left - poly.bounding_rectangle.top_left
445        )
446        return bool(self._mask.overlap(poly._mask, offset))
447
448    def contain(self, coordinate: Coordinate) -> bool:
449        """
450        Checks if a coordinate point lies within this polygon.
451
452        The point is considered inside if it falls within the polygon's
453        bounds, including points on the edges.
454
455        Arguments:
456            `coordinate`: The coordinate point to check.
457
458        Returns:
459            `bool`: Whether the coordinate lies within the polygon.
460
461        Example:
462            ```python
463            triangle = Polygon((Coordinate(0, 0), Coordinate(50, 0),
464                              Coordinate(25, 50)))
465            is_inside = triangle.contain(Coordinate(25, 25))  # True
466            ```
467        """
468        x, y = coordinate - self.bounding_rectangle.top_left
469        width, height = self._mask.get_size()
470        if 0 <= x < width and 0 <= y < height:
471            return bool(self._mask.get_at((x, y)))
472        return False
473
474    @cached_property
475    def length(self) -> Pixel:
476        """
477        Get the perimeter length of the polygon.
478
479        Calculates the total length of all edges in the polygon.
480        For closed polygons, includes the edge from the last point
481        back to the first point.
482
483        Returns:
484            `Pixel`: The perimeter length in pixels.
485
486        Example:
487            ```python
488            triangle = Polygon((Coordinate(0, 0), Coordinate(50, 0),
489                              Coordinate(25, 50)))
490            perimeter = triangle.length  # Total edge length
491            ```
492        """
493        length = sum(
494            p.distance(np) for p, np in zip(self.points, self.points[1:])
495        )
496        if self.closed:
497            length += self.points[0].distance(self.points[-1])
498        return length
499
500    def _draw(
501        self, surf_and_points: Callable[[Surface, tuple[Coordinate, ...]], None]
502    ) -> DrawOnScreen:
503        """
504        Internal method to create a drawing from the polygon.
505
506        Arguments:
507            `surf_and_points`: Function that draws on a surface with points.
508
509        Returns:
510            `DrawOnScreen`: The polygon drawing.
511        """
512        rectangle = self.bounding_rectangle
513        surf = Surface(rectangle.size, SRCALPHA)
514        negated = tuple(p - rectangle.top_left for p in self.points)
515        surf_and_points(surf, negated)
516        return DrawOnScreen(rectangle.top_left, Drawing(surf))

519class Rectangle(Polygon):
520    """
521    A specialized rectangle polygon defined by its top-left corner and size.
522
523    This class provides optimized rectangle operations with efficient
524    collision detection and convenient property access for rectangle
525    edges and corners.
526
527    Arguments:
528        `top_left`: The top-left corner of the rectangle.
529
530        `size`: The dimensions of the rectangle (width and height).
531
532    Example:
533        ```python
534        # Create a 100x50 rectangle at position (10, 20)
535        rect = Rectangle(Coordinate(10, 20), Size(100, 50))
536
537        # Get the center point
538        center = rect.center
539
540        # Check if a point is inside
541        is_inside = rect.contain(Coordinate(50, 40))
542        ```
543    """
544
545    def __init__(self, top_left: Coordinate, size: Size) -> None:
546        """
547        Initialize a rectangle with top-left corner and size.
548
549        Arguments:
550            `top_left`: The top-left corner of the rectangle.
551
552            `size`: The dimensions of the rectangle, including its
553                width and height.
554        """
555        self.top_left = top_left
556        self.size = size
557
558    @property
559    def left(self) -> Pixel:
560        """
561        Gets the leftmost x-coordinate of the rectangle.
562
563        Returns:
564            `Pixel`: The leftmost x-coordinate.

566        return self.top_left.left
567
568    @property

570        """
571        Gets the rightmost x-coordinate of the rectangle.

580        """
581        Gets the topmost y-coordinate of the rectangle.

583        Returns:

587
588    @property
589    def bottom(self) -> Pixel:
590        """
591        Gets the bottommost y-coordinate of the rectangle.

593        Returns:
594            `Pixel`: The bottommost y-coordinate (top + height).

597
598    @property

600        """
601        Gets the top-right coordinate of the rectangle.
602
603        Returns:

610        """

613        Returns:

620        """
621        Gets the bottom-right coordinate of the rectangle.
622
623        Returns:
624            `Coordinate`: The bottom-right coordinate.
625        """
626        return Coordinate(self.right, self.bottom)

630        """
631        Gets the center point of the top edge of the rectangle.
632
633        Returns:
634            `Coordinate`: The top-center coordinate.
635        """
636        return Coordinate(self.left + self.size.width / 2, self.top)
637
638    @property
639    def bottom_center(self) -> Coordinate:
640        """
641        Gets the center point of the bottom edge of the rectangle.
642
643        Returns:
644            `Coordinate`: The bottom-center coordinate.
645        """
646        return Coordinate(self.left + self.size.width / 2, self.bottom)
647
648    @property
649    def center_left(self) -> Coordinate:
650        """
651        Gets the center point of the left edge of the rectangle.
652
653        Returns:
654            `Coordinate`: The left-center coordinate.
655        """
656        return Coordinate(self.left, self.top + self.size.height / 2)
657
658    @property
659    def center_right(self) -> Coordinate:
660        """
661        Gets the center point of the right edge of the rectangle.
662
663        Returns:
664            `Coordinate`: The right-center coordinate.
665        """
666        return Coordinate(self.right, self.top + self.size.height / 2)
667
668    @property
669    def center(self) -> Coordinate:
670        """
671        Gets the center point of the rectangle.
672
673        Returns:
674            `Coordinate`: The center coordinate of the rectangle.
675        """
676        width, height = self.size
677        return Coordinate(self.left + width / 2, self.top + height / 2)
678
679    @property
680    def points(self) -> tuple[Coordinate, ...]:
681        """
682        Get the coordinates of the corners of the rectangle.
683
684        Returns the four corner points in clockwise order starting
685        from the top-left corner.
686
687        Returns:
688            `tuple[Coordinate, ...]`: The coordinates of the corners
689                of the rectangle.
690        """
691        return (
692            self.top_left,
693            self.top_right,
694            self.bottom_right,
695            self.bottom_left,
696        )
697
698    @override
699    def collide(self, poly: Polygon) -> bool:
700        if not isinstance(poly, Rectangle):
701            return super().collide(poly)
702
703        return (
704            self.top_left.left < poly.top_right.left
705            and self.top_right.left > poly.top_left.left
706            and self.top_left.top < poly.bottom_right.top
707            and self.bottom_right.top > poly.top_left.top
708        )
709
710    @override
711    def contain(self, coordinate: Coordinate) -> bool:
712        return (
713            self.left < coordinate.left < self.right
714            and self.top < coordinate.top < self.bottom
715        )
716
717    def __add__(self, coordinate: Coordinate) -> Self:
718        """
719        Shift the rectangle by the specified coordinate.
720
721        Arguments:
722            `coordinate`: The coordinate to shift the rectangle by.
723
724        Returns:
725            `Rectangle`: A new rectangle shifted by the coordinate.
726        """
727        return Rectangle(self.top_left + coordinate, self.size)

729    def fill(
730        self, color: Rgba, border_radius: Pixel | None = None
731    ) -> DrawOnScreen:
732        """
733        Creates a filled rectangle drawing with the provided color.
734
735        Arguments:
736            `color`: The color to fill the rectangle with.
737
738            `border_radius`: Optional border radius for rounded corners.
739                If `None`, creates a rectangle with sharp corners.
740
741        Returns:
742            `DrawOnScreen`: A drawing of the filled rectangle.
743
744        Example:
745            ```python
746            rect = Rectangle(Coordinate(0, 0), Size(100, 50))
747            filled = rect.fill(Rgba(255, 0, 0, 255))  # Red rectangle
748            rounded = rect.fill(Rgba(0, 255, 0, 255), border_radius=10)
749            ```
750        """
751        surf = Surface(self.size, SRCALPHA)
752        rectangle = Rect(Coordinate(0, 0), self.size)
753        rect(surf, color, rectangle, border_radius=border_radius or -1)
754        return DrawOnScreen(self.top_left, Drawing(surf))
