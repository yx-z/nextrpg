 23class Coordinate(namedtuple("Coordinate", "left top")):
 24    """
 25    Represents a 2D coordinate with immutability and provides methods
 26    for scaling and shifting coordinates.
 27
 28    Attributes:
 29        `left`: The horizontal position of the coordinate, measured by
 30            the number of pixels from the left edge of the game window.
 31
 32        `top`: The vertical position of the coordinate, measured by
 33            the number of pixels from the top edge of the game window.
 34    """
 35
 36    left: Pixel
 37    top: Pixel
 38
 39    def __sub__(self, other: Coordinate) -> Coordinate:
 40        return Coordinate(self.left - other.left, self.top - other.top)
 41
 42    def __add__(self, offset: DirectionalOffset | Coordinate) -> Coordinate:
 43        """
 44        Shifts the coordinate in the specified direction by a given offset.
 45        Supports both orthogonal and diagonal directions.
 46
 47        Or add two coordinates together.
 48
 49        For diagonal directions, the offset is divided proportionally.
 50        For example, an offset of `sqrt(2)` in `UP_LEFT` direction shifts
 51        the coordinate `Pixel(1)` in both `UP` and `LEFT` directions.
 52
 53        Args:
 54            `offset`: A `DirectionalOffset` representing the direction
 55                and offset, or `Coordinate` to add to the current `Coordinate`.
 56
 57        Returns:
 58            `Coordinate`: A new coordinate shifted by the specified offset in
 59            the given direction.
 60        """
 61        if isinstance(offset, Coordinate):
 62            return Coordinate(self.left + offset.left, self.top + offset.top)
 63
 64        match offset.direction:
 65            case Direction.UP:
 66                return Coordinate(self.left, self.top - offset.offset)
 67            case Direction.DOWN:
 68                return Coordinate(self.left, self.top + offset.offset)
 69            case Direction.LEFT:
 70                return Coordinate(self.left - offset.offset, self.top)
 71            case Direction.RIGHT:
 72                return Coordinate(self.left + offset.offset, self.top)
 73
 74        diag = offset.offset / sqrt(2)
 75        match offset.direction:
 76            case Direction.UP_LEFT:
 77                return Coordinate(self.left - diag, self.top - diag)
 78            case Direction.UP_RIGHT:
 79                return Coordinate(self.left + diag, self.top - diag)
 80            case Direction.DOWN_LEFT:
 81                return Coordinate(self.left - diag, self.top + diag)
 82            case Direction.DOWN_RIGHT:
 83                return Coordinate(self.left + diag, self.top + diag)
 84        return NotImplemented

212    def rectangle(self) -> Rectangle:
213        """
214        Gets the rectangular bounds of the drawing on screen.
215
216        Returns:
217            `Rectangle`: A rectangle defining the drawing's position and size
218            on screen.
219        """
220        return Rectangle(self.top_left, self.drawing.size)
221
222    @cached_property
223    def visible_rectangle(self) -> Rectangle:
224        """
225        Calculate the actual visible bounds of the drawing,
226        ignoring transparent pixels.
227
228        Returns:
229            `Rectangle`: A rectangle defining only the visible (non-transparent)
230            portion of the drawing on screen.
231        """
232        visible = [
233            Coordinate(x, y)
234            for x, y in product(
235                range(ceil(self.drawing.width)),
236                range(ceil(self.drawing.height)),
237            )
238            # `drawing._surface` to ignore the debug background.
239            if self.drawing._surface.get_at((x, y)).a
240        ]
241        if not visible:
242            return Rectangle(Coordinate(0, 0), Size(0, 0))
243
244        min_x = min(c.left for c in visible)
245        min_y = min(c.top for c in visible)
246        max_x = max(c.left for c in visible)
247        max_y = max(c.top for c in visible)
248        return Rectangle(
249            self.top_left + Coordinate(min_x, min_y),
250            Size(max_x - min_x, max_y - min_y),
251        )
252
253    @cached_property
254    def pygame(self) -> tuple[Surface, tuple[Pixel, Pixel]]:
255        """
256        Gets the pygame surface and coordinate tuple for rendering.
257
258        Returns:
259            `tuple[pygame.Surface, tuple[float, float]]`:
260                A tuple containing the `pygame.Surface` and a tuple of the left
261                and top coordinates (x, y).
262        """
263        return self.drawing.pygame, self.top_left

266        """
267        Shift the drawing by the specified coordinate.
268
269        Args:
270            `coord`: The coordinate to shift the drawing by.
271
272        Returns:
273            `DrawOnScreen`: A new `DrawOnScreen` shifted by the coordinate.
274        """
275        return DrawOnScreen(self.top_left + coord, self.drawing)

278class Polygon(Protocol):
279    """
280    Abstract base class for polygon, defining the common interface/behavior.
281    """
282
283    @cached_property
284    def points(self) -> tuple[Coordinate, ...]:
285        """
286        Get the points of the polygon.
287
288        Returns:
289            `tuple[Coordinate, ...]`: A tuple of `Coordinate` objects
290                representing the points.
291        """
292
293    @cached_property
294    def bounding_rectangle(self) -> Rectangle:
295        """
296        Get a rectangle bounding the polygon.
297
298        Returns:
299            `Rectangle`: A rectangle bounding the polygon.
300        """
301        min_x = min(c.left for c in self.points)
302        min_y = min(c.top for c in self.points)
303        max_x = max(c.left for c in self.points)
304        max_y = max(c.top for c in self.points)
305        return Rectangle(
306            Coordinate(min_x, min_y), Size(max_x - min_x, max_y - min_y)
307        )
308
309    @cached_property
310    def _mask(self) -> Mask:
311        return from_surface(self.fill(Rgba(0, 0, 0, 255)).drawing.pygame)
312
313    def fill(self, color: Rgba) -> DrawOnScreen:
314        """
315        Creates a colored `DrawOnScreen` with the provided color.
316
317        Args:
318            `Color`: The color to fill the rectangle.
319
320        Returns:
321            `DrawOnScreen`: A transparent surface matching rectangle dimensions.
322        """
323        rect = self.bounding_rectangle
324        surface = Surface(rect.size, SRCALPHA)
325        polygon(surface, color, [(p - rect.top_left) for p in self.points])
326        return DrawOnScreen(rect.top_left, Drawing(surface))
327
328    def collide(self, poly: Polygon) -> bool:
329        """
330        Checks if this rectangle overlaps with another polygon.
331
332        Args:
333            `poly`: The polygon to check for collision with.
334
335        Returns:
336            `bool`: True if two polygons overlap, False otherwise.
337        """
338        if not self.bounding_rectangle.collide(poly.bounding_rectangle):
339            return False
340        offset = (
341            self.bounding_rectangle.top_left - poly.bounding_rectangle.top_left
342        )
343        return bool(self._mask.overlap(poly._mask, offset))
344
345    def contain(self, coordinate: Coordinate) -> bool:
346        """
347        Checks if a coordinate point lies within this polygon.
348
349        The point is considered inside if it falls in the polygon's bounds,
350        including points on the edges.
351
352        Arguments:
353            `coordinate`: The coordinate point to check
354
355        Returns:
356            `bool`: Whether the coordinate lies within the rectangle.
357        """
358        x, y = coordinate - self.bounding_rectangle.top_left
359        width, height = self._mask.get_size()
360        if 0 <= x < width and 0 <= y < height:
361            return bool(self._mask.get_at((x, y)))
362        return False
