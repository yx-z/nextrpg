 26class Coordinate(namedtuple("Coordinate", "left top")):
 27    """
 28    Represents a 2D coordinate with immutability and provides methods
 29    for scaling and shifting coordinates.
 30
 31    Attributes:
 32        `left`: The horizontal position of the coordinate, measured by
 33            the number of pixels from the left edge of the game window.
 34
 35        `top`: The vertical position of the coordinate, measured by
 36            the number of pixels from the top edge of the game window.
 37    """
 38
 39    left: Pixel
 40    top: Pixel
 41
 42    @override
 43    def __sub__(self, offset: Self) -> Self:
 44        """
 45        Subtracts a coordinate from this coordinate.
 46
 47        Arguments:
 48            `offset`: Coordinate to subtract from the current coordinate.
 49
 50        Returns:
 51            `Coordinate`: A new coordinate shifted by the specified offset.
 52        """
 53        return Coordinate(self.left - offset.left, self.top - offset.top)
 54
 55    def shift(self, offset: DirectionalOffset | Self) -> Self:
 56        """
 57        Shifts the coordinate in the specified direction by a given offset.
 58        Supports both orthogonal and diagonal directions.
 59
 60        Or add two coordinates together.
 61
 62        For diagonal directions, the offset is divided proportionally.
 63        For example, an offset of `sqrt(2)` in `UP_LEFT` direction shifts
 64        the coordinate `Pixel(1)` in both `UP` and `LEFT` directions.
 65
 66        Arguments:
 67            `offset`: A `DirectionalOffset` representing the direction
 68                and offset, or `Coordinate` to add to the current `Coordinate`.
 69
 70        Returns:
 71            `Coordinate`: A new coordinate shifted by the specified offset in
 72            the given direction.
 73        """
 74        if isinstance(offset, Coordinate):
 75            return Coordinate(self.left + offset.left, self.top + offset.top)
 76
 77        match offset.direction:
 78            case Direction.UP:
 79                return Coordinate(self.left, self.top - offset.offset)
 80            case Direction.DOWN:
 81                return Coordinate(self.left, self.top + offset.offset)
 82            case Direction.LEFT:
 83                return Coordinate(self.left - offset.offset, self.top)
 84            case Direction.RIGHT:
 85                return Coordinate(self.left + offset.offset, self.top)
 86
 87        diag = offset.offset / sqrt(2)
 88        match offset.direction:
 89            case Direction.UP_LEFT:
 90                return Coordinate(self.left - diag, self.top - diag)
 91            case Direction.UP_RIGHT:
 92                return Coordinate(self.left + diag, self.top - diag)
 93            case Direction.DOWN_LEFT:
 94                return Coordinate(self.left - diag, self.top + diag)
 95            case Direction.DOWN_RIGHT:
 96                return Coordinate(self.left + diag, self.top + diag)
 97
 98    def __repr__(self) -> str:
 99        return f"({self.left:.1f}, {self.top:.1f})"

217class DrawOnScreen:
218    """
219    Represents a drawable element positioned on the screen with its coordinates.
220
221    This immutable class combines a drawing with its position and provides
222    properties to access various coordinates and dimensions of the drawing
223    on the screen.
224
225    Attributes:
226        `top_left`: The top-left position of the drawing on the screen.
227
228        `drawing`: The drawable element to be displayed on the screen.
229    """
230
231    top_left: Coordinate
232    drawing: Drawing
233
234    @property
235    def rectangle(self) -> Rectangle:
236        """
237        Gets the rectangular bounds of the drawing on screen.
238
239        Returns:
240            `Rectangle`: A rectangle defining the drawing's position and size
241            on screen.
242        """
243        return Rectangle(self.top_left, self.drawing.size)
244
245    @cached_property
246    def visible_rectangle(self) -> Rectangle:
247        """
248        Calculate the actual visible bounds of the drawing,
249        ignoring transparent pixels.
250
251        Returns:
252            `Rectangle`: A rectangle defining only the visible (non-transparent)
253            portion of the drawing on screen.
254        """
255        visible = [
256            Coordinate(x, y)
257            for x, y in product(
258                range(ceil(self.drawing.width)),
259                range(ceil(self.drawing.height)),
260            )
261            # `drawing._surface` to ignore the debug background.
262            if self.drawing._surface.get_at((x, y)).a
263        ]
264        if not visible:
265            return Rectangle(Coordinate(0, 0), Size(0, 0))
266
267        min_x = min(c.left for c in visible)
268        min_y = min(c.top for c in visible)
269        max_x = max(c.left for c in visible)
270        max_y = max(c.top for c in visible)
271        return Rectangle(
272            self.top_left.shift(Coordinate(min_x, min_y)),
273            Size(max_x - min_x, max_y - min_y),
274        )
275
276    @property
277    def pygame(self) -> tuple[Surface, Coordinate]:
278        """
279        Gets the pygame surface and coordinate tuple for rendering.
280
281        Returns:
282            `tuple[pygame.Surface, tuple[float, float]]`:
283                A tuple containing the `pygame.Surface` and a tuple of the left
284                and top coordinates (x, y).
285        """
286        return self.drawing.pygame, self.top_left
287
288    def shift(self, coord: Coordinate) -> Self:
289        """
290        Shift the drawing by the specified coordinate.
291
292        Arguments:
293            `coord`: The coordinate to shift the drawing by.
294
295        Returns:
296            `DrawOnScreen`: A new `DrawOnScreen` shifted by the coordinate.
297        """
298        return DrawOnScreen(self.top_left.shift(coord), self.drawing)

302class Polygon:
303    """
304    Polygon is a sequence of points.
305    """
306
307    points: tuple[Coordinate, ...]
308
309    @cached_property
310    def bounding_rectangle(self) -> Rectangle:
311        """
312        Get a rectangle bounding the polygon.
313
314        Returns:
315            `Rectangle`: A rectangle bounding the polygon.
316        """
317        min_x = min(c.left for c in self.points)
318        min_y = min(c.top for c in self.points)
319        max_x = max(c.left for c in self.points)
320        max_y = max(c.top for c in self.points)
321        return Rectangle(
322            Coordinate(min_x, min_y), Size(max_x - min_x, max_y - min_y)
323        )
324
325    @cached_property
326    def _mask(self) -> Mask:
327        return from_surface(self.fill(Rgba(0, 0, 0, 255)).drawing.pygame)
328
329    def fill(self, color: Rgba) -> DrawOnScreen:
330        """
331        Creates a colored `DrawOnScreen` with the provided color.
332
333        Arguments:
334            `Color`: The color to fill the rectangle.
335
336        Returns:
337            `DrawOnScreen`: A transparent surface matching rectangle dimensions.
338        """
339        rect = self.bounding_rectangle
340        surface = Surface(rect.size, SRCALPHA)
341        polygon(surface, color, [(p - rect.top_left) for p in self.points])
342        return DrawOnScreen(rect.top_left, Drawing(surface))
343
344    def collide(self, poly: Self) -> bool:
345        """
346        Checks if this rectangle overlaps with another polygon.
347
348        Arguments:
349            `poly`: The polygon to check for collision with.
350
351        Returns:
352            `bool`: True if two polygons overlap, False otherwise.
353        """
354        if not self.bounding_rectangle.collide(poly.bounding_rectangle):
355            return False
356        offset = (
357            self.bounding_rectangle.top_left - poly.bounding_rectangle.top_left
358        )
359        return bool(self._mask.overlap(poly._mask, offset))
360
361    def contain(self, coordinate: Coordinate) -> bool:
362        """
363        Checks if a coordinate point lies within this polygon.
364
365        The point is considered inside if it falls in the polygon's bounds,
366        including points on the edges.
367
368        Arguments:
369            `coordinate`: The coordinate point to check
370
371        Returns:
372            `bool`: Whether the coordinate lies within the rectangle.
373        """
374        x, y = coordinate - self.bounding_rectangle.top_left
375        width, height = self._mask.get_size()
376        if 0 <= x < width and 0 <= y < height:
377            return bool(self._mask.get_at((x, y)))
378        return False
