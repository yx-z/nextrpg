nextrpg.draw_on_screen

Drawable elements and rendering utilities for NextRPG.

This module provides the core drawing system for NextRPG games. It defines classes for representing drawable elements, managing their positioning on screen, and handling various drawing operations.

The module includes:

  • Drawing: Represents a drawable element with size and surface access
  • DrawOnScreen: Combines a drawing with its position on screen
  • Polygon: Represents polygon shapes with collision detection
  • Rectangle: Specialized rectangle with optimized collision detection

These classes provide the foundation for all visual elements in NextRPG games, from sprites and backgrounds to UI elements and geometric shapes.

Example:
from nextrpg.draw_on_screen import Drawing, DrawOnScreen, Rectangle
from nextrpg.coordinate import Coordinate
from nextrpg.core import Size, Rgba

# Load a sprite
sprite = Drawing("assets/player.png")

# Position it on screen
player = DrawOnScreen(Coordinate(100, 100), sprite)

# Create a colored rectangle
rect = Rectangle(Coordinate(0, 0), Size(50, 50))
colored_rect = rect.fill(Rgba(255, 0, 0, 255))

 63class Drawing:
 64    """
 65    Represents a drawable element and provides methods for accessing its size
 66    and dimensions.
 67

 69    `pygame.Surface` as input. It provides properties to access surface
 70    dimensions and size and methods to crop and scale the surface.
 71

 74            or a `pygame.Surface` object.
 75

 77        ```python
 78        # Load from file
 79        sprite = Drawing("assets/player.png")
 80
 81        # Use existing surface
 82        surface = pygame.Surface((32, 32))
 83        drawing = Drawing(surface)
 84
 85        # Get dimensions
 86        size = drawing.size  # Returns Size(width, height)
 87        width = drawing.width  # Returns int
 88        ```

133    def crop(self, top_left: Coordinate, size: Size) -> Self:
134        """
135        Crops a rectangular portion of the drawing specified by the
136        top-left corner and the size.
137
138        The method extracts a subsection of the drawing based on the provided
139        coordinates and dimensions and returns a new `Drawing` instance.
140        The original drawing remains unchanged.
141
142        Arguments:
143            `top_left`: The top-left coordinate of the rectangle to be cropped.
144
145            `size`: The width and height of the rectangle to be cropped.
146
147        Returns:
148            `Drawing`: A new `Drawing` instance representing the cropped area.
149
150        Example:
151            ```python
152            drawing = Drawing("spritesheet.png")
153            # Crop a 32x32 sprite from position (64, 0)
154            sprite = drawing.crop(Coordinate(64, 0), Size(32, 32))
155            ```
156        """
157        left, top = top_left
158        width, height = size

181    @cached_property
182    def visible_rectangle(self) -> Rectangle:
183        """
184        Get the bounding rectangle of visible (non-transparent) pixels.
185
186        Calculates the smallest rectangle that contains all non-transparent
187        pixels in the drawing. This is useful for collision detection
188        and optimizing drawing operations.
189
190        Returns:
191            `Rectangle`: The bounding rectangle of visible pixels.
192
193        Example:
194            ```python
195            drawing = Drawing("sprite.png")
196            visible_bounds = drawing.visible_rectangle
197            # Use for collision detection
198            ```
199        """
200        visible = [
201            Coordinate(x, y)
202            for x in range(ceil(self.width))
203            for y in range(ceil(self.height))
204            if self._surface.get_at((x, y)).a
205        ]
206        if not visible:
207            return Rectangle(Coordinate(0, 0), Size(0, 0))
208
209        min_x = min(c.left for c in visible)
210        min_y = min(c.top for c in visible)
211        max_x = max(c.left for c in visible)
212        max_y = max(c.top for c in visible)
213        return Rectangle(
214            Coordinate(min_x, min_y), Size(max_x - min_x, max_y - min_y)
215        )

217    @cached_property
218    def _debug_surface(self) -> Surface | None:
219        """
220        Get debug surface with background color if debug mode is enabled.
221
222        Returns:
223            `Surface | None`: Debug surface or None if debug mode is disabled.
224        """
225        if (debug := config().debug) and (
226            color := debug.drawing_background_color
227        ):
228            surface = Surface(self.size, SRCALPHA)
229            surface.fill(color)

248        logger.debug(t"Loading {self.resource}")

254class DrawOnScreen:
255    """
256    Represents a drawable element positioned on the screen with its coordinates.
257
258    This immutable class combines a drawing with its position and provides
259    properties to access various coordinates and dimensions of the drawing
260    on the screen.
261
262    Arguments:
263        `top_left`: The top-left position of the drawing on the screen.
264

266
267    Example:
268        ```python
269        sprite = Drawing("player.png")
270        player = DrawOnScreen(Coordinate(100, 100), sprite)
271
272        # Get the rectangle bounds
273        bounds = player.rectangle
274
275        # Check if point is inside
276        is_inside = bounds.contain(Coordinate(120, 120))
277        ```
278    """
279

281    drawing: Drawing
282
283    @property
284    def rectangle(self) -> Rectangle:
285        """
286        Gets the rectangular bounds of the drawing on screen.
287
288        Returns:
289            `Rectangle`: A rectangle defining the drawing's position and size
290            on screen.
291        """
292        return Rectangle(self.top_left, self.drawing.size)

354        return DrawOnScreen(self.top_left, self.drawing.set_alpha(alpha))

359class Polygon:
360    """
361    Represents a polygon shape defined by a sequence of points.
362
363    This class provides polygon functionality including collision detection,
364    drawing operations, and geometric calculations. Polygons can be either
365    closed (default) or open shapes.
366
367    Arguments:
368        `points`: A tuple of coordinates defining the polygon vertices.
369
370        `closed`: Whether the polygon is closed (last point connects to first).
371            Defaults to `True`.
372
373    Example:
374        ```python
375        # Create a triangle
376        triangle = Polygon((
377            Coordinate(0, 0),
378            Coordinate(50, 0),
379            Coordinate(25, 50)
380        ))
381
382        # Create an open line
383        line = Polygon((
384            Coordinate(0, 0),
385            Coordinate(100, 100)
386        ), closed=False)
387        ```
388    """
389
390    points: tuple[Coordinate, ...]
391    closed: bool = True
392
393    @cached_property
394    def bounding_rectangle(self) -> Rectangle:
395        """
396        Get a rectangle bounding the polygon.
397
398        Calculates the smallest rectangle that completely contains
399        all points of the polygon.
400
401        Returns:
402            `Rectangle`: A rectangle bounding the polygon.
403
404        Example:
405            ```python
406            polygon = Polygon((Coordinate(0, 0), Coordinate(50, 50)))
407            bounds = polygon.bounding_rectangle
408            ```
409        """
410        min_x = min(c.left for c in self.points)
411        min_y = min(c.top for c in self.points)
412        max_x = max(c.left for c in self.points)
413        max_y = max(c.top for c in self.points)
414        return Rectangle(
415            Coordinate(min_x, min_y), Size(max_x - min_x, max_y - min_y)
416        )
417
418    @cached_property
419    def _mask(self) -> Mask:
420        """
421        Get the collision mask for this polygon.
422
423        Creates a pygame mask from the filled polygon for precise
424        collision detection.
425
426        Returns:
427            `Mask`: The collision mask for this polygon.
428        """
429        return from_surface(self.fill(BLACK).drawing.pygame)
430
431    def fill(self, color: Rgba) -> DrawOnScreen:
432        """
433        Creates a filled polygon drawing with the provided color.
434
435        Arguments:
436            `color`: The color to fill the polygon with.
437
438        Returns:
439            `DrawOnScreen`: A drawing of the filled polygon.
440
441        Example:
442            ```python
443            triangle = Polygon((Coordinate(0, 0), Coordinate(50, 0),
444                              Coordinate(25, 50)))
445            filled = triangle.fill(Rgba(255, 0, 0, 255))  # Red triangle
446            ```
447        """
448        return self._draw(lambda surf, points: polygon(surf, color, points))
449
450    def line(
451        self, color: Rgba, stroke_width: Pixel | None = None
452    ) -> DrawOnScreen:
453        """
454        Creates a line drawing of the polygon with the provided color.
455
456        Arguments:
457            `color`: The color of the line.
458
459            `stroke_width`: The width of the line in pixels.
460                If `None`, uses the default stroke width from config.
461
462        Returns:
463            `DrawOnScreen`: A drawing of the polygon outline.
464
465        Example:
466            ```python
467            triangle = Polygon((Coordinate(0, 0), Coordinate(50, 0),
468                              Coordinate(25, 50)))
469            outline = triangle.line(Rgba(0, 0, 255, 255), stroke_width=2)
470            ```
471        """
472        stroke = (
473            config().draw_on_screen.stroke_width
474            if stroke_width is None
475            else stroke_width
476        )
477        return self._draw(
478            lambda surf, points: lines(surf, color, self.closed, points, stroke)
479        )
480
481    def collide(self, poly: Self) -> bool:
482        """
483        Checks if this polygon overlaps with another polygon.
484
485        Performs precise collision detection using pygame masks.
486        First checks bounding rectangles for efficiency, then
487        performs pixel-perfect collision detection.
488
489        Arguments:
490            `poly`: The polygon to check for collision with.
491
492        Returns:
493            `bool`: `True` if the polygons overlap, `False` otherwise.
494
495        Example:
496            ```python
497            poly1 = Polygon((Coordinate(0, 0), Coordinate(50, 50)))
498            poly2 = Polygon((Coordinate(25, 25), Coordinate(75, 75)))
499            is_colliding = poly1.collide(poly2)  # True
500            ```
501        """
502        if not self.bounding_rectangle.collide(poly.bounding_rectangle):
503            return False
504        offset = self.bounding_rectangle.top_left.shift(
505            poly.bounding_rectangle.top_left.negate
506        )
507        return bool(self._mask.overlap(poly._mask, offset))
508
509    def contain(self, coordinate: Coordinate) -> bool:
510        """
511        Checks if a coordinate point lies within this polygon.
512
513        The point is considered inside if it falls within the polygon's
514        bounds, including points on the edges.
515
516        Arguments:
517            `coordinate`: The coordinate point to check.
518
519        Returns:
520            `bool`: Whether the coordinate lies within the polygon.
521
522        Example:
523            ```python
524            triangle = Polygon((Coordinate(0, 0), Coordinate(50, 0),
525                              Coordinate(25, 50)))
526            is_inside = triangle.contain(Coordinate(25, 25))  # True
527            ```
528        """
529        x, y = coordinate.shift(self.bounding_rectangle.top_left.negate)
530        width, height = self._mask.get_size()
531        if 0 <= x < width and 0 <= y < height:
532            return bool(self._mask.get_at((x, y)))
533        return False
534
535    @cached_property
536    def length(self) -> Pixel:
537        """
538        Get the perimeter length of the polygon.
539
540        Calculates the total length of all edges in the polygon.
541        For closed polygons, includes the edge from the last point
542        back to the first point.
543
544        Returns:
545            `Pixel`: The perimeter length in pixels.
546
547        Example:
548            ```python
549            triangle = Polygon((Coordinate(0, 0), Coordinate(50, 0),
550                              Coordinate(25, 50)))
551            perimeter = triangle.length  # Total edge length
552            ```
553        """
554        length = sum(
555            p.distance(np) for p, np in zip(self.points, self.points[1:])
556        )
557        if self.closed:
558            length += self.points[0].distance(self.points[-1])
559        return length
560
561    def _draw(
562        self, surf_and_points: Callable[[Surface, tuple[Coordinate, ...]], None]
563    ) -> DrawOnScreen:
564        """
565        Internal method to create a drawing from the polygon.
566
567        Arguments:
568            `surf_and_points`: Function that draws on a surface with points.
569
570        Returns:
571            `DrawOnScreen`: The polygon drawing.
572        """
573        rectangle = self.bounding_rectangle
574        surf = Surface(rectangle.size, SRCALPHA)
575        negated = tuple(
576            (p.shift(rectangle.top_left.negate)) for p in self.points
577        )
578        surf_and_points(surf, negated)
579        return DrawOnScreen(rectangle.top_left, Drawing(surf))

743    @cached_property
