 21class DrawOnScreen:
 22    """
 23    Represents a drawable element positioned on the screen with its coordinates.
 24
 25    This immutable class combines a drawing with its position and provides
 26    properties to access various coordinates and dimensions of the drawing
 27    on the screen.
 28
 29    Attributes:
 30        `top_left`: The top-left position of the drawing on the screen.
 31
 32        `drawing`: The drawable element to be displayed on the screen.
 33    """
 34
 35    top_left: Coordinate
 36    drawing: Drawing
 37
 38    @property
 39    def rectangle(self) -> Rectangle:
 40        """
 41        Gets the rectangular bounds of the drawing on screen.
 42
 43        Returns:
 44            `Rectangle`: A rectangle defining the drawing's position and size
 45            on screen.
 46        """
 47        return Rectangle(self.top_left, self.drawing.size)
 48
 49    @cached_property
 50    def visible_rectangle(self) -> Rectangle:
 51        """
 52        Calculate the actual visible bounds of the drawing,
 53        ignoring transparent pixels.
 54
 55        Returns:
 56            `Rectangle`: A rectangle defining only the visible (non-transparent)
 57            portion of the drawing on screen.
 58        """
 59        visible = [
 60            Coordinate(x, y)
 61            for x, y in product(
 62                range(ceil(self.drawing.width)),
 63                range(ceil(self.drawing.height)),
 64            )
 65            # `drawing._surface` to ignore the debug background.
 66            if self.drawing._surface.get_at((x, y)).a
 67        ]
 68        if not visible:
 69            return Rectangle(Coordinate(0, 0), Size(0, 0))
 70
 71        min_x = min(c.left for c in visible)
 72        min_y = min(c.top for c in visible)
 73        max_x = max(c.left for c in visible)
 74        max_y = max(c.top for c in visible)
 75        return Rectangle(
 76            self.top_left.shift(Coordinate(min_x, min_y)),
 77            Size(max_x - min_x, max_y - min_y),
 78        )
 79
 80    @property
 81    def pygame(self) -> tuple[Surface, Coordinate]:
 82        """
 83        Gets the pygame surface and coordinate tuple for rendering.
 84
 85        Returns:
 86            `tuple[pygame.Surface, tuple[float, float]]`:
 87                A tuple containing the `pygame.Surface` and a tuple of the left
 88                and top coordinates (x, y).
 89        """
 90        return self.drawing.pygame, self.top_left
 91
 92    def shift(self, coord: Coordinate) -> Self:
 93        """
 94        Shift the drawing by the specified coordinate.
 95
 96        Arguments:
 97            `coord`: The coordinate to shift the drawing by.
 98
 99        Returns:
100            `DrawOnScreen`: A new `DrawOnScreen` shifted by the coordinate.
101        """
102        return DrawOnScreen(self.top_left.shift(coord), self.drawing)

106class Polygon:
107    """
108    Polygon is a sequence of points.
109    """
110
111    points: tuple[Coordinate, ...]
112    is_closed: bool = True
113
114    @cached_property
115    def bounding_rectangle(self) -> Rectangle:
116        """
117        Get a rectangle bounding the polygon.
118
119        Returns:
120            `Rectangle`: A rectangle bounding the polygon.
121        """
122        min_x = min(c.left for c in self.points)
123        min_y = min(c.top for c in self.points)
124        max_x = max(c.left for c in self.points)
125        max_y = max(c.top for c in self.points)
126        return Rectangle(
127            Coordinate(min_x, min_y), Size(max_x - min_x, max_y - min_y)
128        )
129
130    @cached_property
131    def _mask(self) -> Mask:
132        return from_surface(self.fill(Rgba(0, 0, 0, 255)).drawing.pygame)
133
134    def fill(self, color: Rgba) -> DrawOnScreen:
135        """
136        Creates a colored `DrawOnScreen` with the provided color.
137
138        Arguments:
139            `Color`: The color to fill the rectangle.
140
141        Returns:
142            `DrawOnScreen`: A transparent surface matching rectangle dimensions.
143        """
144        rect = self.bounding_rectangle
145        surf = Surface(rect.size, SRCALPHA)
146        negated = [(p.shift(rect.top_left.negate)) for p in self.points]
147        polygon(surf, color, negated)
148        return DrawOnScreen(rect.top_left, Drawing(surf))
149
150    def collide(self, poly: Self) -> bool:
151        """
152        Checks if this rectangle overlaps with another polygon.
153
154        Arguments:
155            `poly`: The polygon to check for collision with.
156
157        Returns:
158            `bool`: True if two polygons overlap, False otherwise.
159        """
160        if not self.bounding_rectangle.collide(poly.bounding_rectangle):
161            return False
162        offset = self.bounding_rectangle.top_left.shift(
163            poly.bounding_rectangle.top_left.negate
164        )
165        return bool(self._mask.overlap(poly._mask, offset))
166
167    def contain(self, coordinate: Coordinate) -> bool:
168        """
169        Checks if a coordinate point lies within this polygon.
170
171        The point is considered inside if it falls in the polygon's bounds,
172        including points on the edges.
173
174        Arguments:
175            `coordinate`: The coordinate point to check
176
177        Returns:
178            `bool`: Whether the coordinate lies within the rectangle.
179        """
180        x, y = coordinate.shift(self.bounding_rectangle.top_left.negate)
181        width, height = self._mask.get_size()
182        if 0 <= x < width and 0 <= y < height:
183            return bool(self._mask.get_at((x, y)))
184        return False

340    def collide(self, poly: Polygon) -> bool:
341        if isinstance(poly, Rectangle):
342            return (
343                self.top_left.left < poly.top_right.left
344                and self.top_right.left > poly.top_left.left
345                and self.top_left.top < poly.bottom_right.top
346                and self.bottom_right.top > poly.top_left.top
347            )
348        return super().collide(poly)
349
350    def contain(self, coordinate: Coordinate) -> bool:
351        return (
352            self.left < coordinate.left < self.right
353            and self.top < coordinate.top < self.bottom
354        )
