nextrpg.npcs

NPC (Non-Player Character) system for nextrpg.

This module provides the core NPC functionality for the nextrpg game engine. It includes classes for managing NPCs on screen, event specifications, and scene management for player-NPC interactions.

Features:
  • NpcOnScreen: In-game NPC interface for stationary NPCs
  • EventfulScene: Scene that supports event execution via coroutines/generators
  • RpgEventScene: Scene wrapper for RPG events
  • NpcSpec: Base class for NPC specifications
  • Type aliases for event specifications and generators

108@export
109@dataclass(frozen=True)
110@event_as_attr
111class EventfulScene(Scene):
112    """Scene that supports event execution via coroutines/generators.
113
114    This scene type allows for complex event sequences to be executed
115    using Python generators. It manages the state of ongoing events
116    and provides methods for event continuation and completion.
117
118    Attributes:
119        player: The player character on screen.
120        npcs: Tuple of NPCs present in the scene.
121        npc: Currently active NPC (if any).
122        _event_generator: Current event generator being executed.
123        _event_result: Result from the previous event step.
124
125    Example:
126        ```python
127        scene = EventfulScene(
128            player=player,
129            npcs=(npc1, npc2)
130        )
131
132        # Handle events
133        scene = scene.event(key_press_event)
134
135        # Update scene over time
136        scene = scene.tick(time_delta)
137        ```
138    """
139
140    player: PlayerOnScreen
141    npcs: tuple[NpcOnScreen, ...] = field(default_factory=tuple)
142    _: KW_ONLY = not_constructor_below()
143    npc: NpcOnScreen | None = None
144    _event_generator: RpgEventGenerator | None = None
145    _event_result: Any = None
146
147    @cached_property
148    def npc_dict(self) -> dict[str, NpcOnScreen]:
149        """Get a dictionary mapping NPC names to NPC objects.
150
151        Returns:
152            Dictionary with NPC object names as keys and NPC objects as values.
153        """
154        return {n.spec.object_name: n for n in self.npcs}
155
156    def event(self, event: PygameEvent) -> Scene:
157        """Handle pygame events and trigger NPC interactions.
158
159        This method processes pygame events and automatically triggers
160        NPC interactions when the player collides with an NPC and
161        presses the confirm key.
162
163        Args:
164            event: The pygame event to process.
165
166        Returns:
167            The updated scene after processing the event.
168        """
169        if (
170            not self.npc
171            and isinstance(event, KeyPressDown)
172            and event.key is KeyboardKey.CONFIRM
173            and (npc := self._collided_npc)
174        ):
175            logger.debug(t"Collided with {npc.spec.object_name}")
176            scene = self._trigger(npc)
177            generator = scene.npc.generator(scene.player, scene.npc, scene)
178            logger.debug(t"Event {generator.__name__} started.")
179            return next(generator)(generator, scene)
180        return self.event_without_npc_trigger(event)
181
182    def event_without_npc_trigger(self, event: PygameEvent) -> Self:
183        """Handle events without triggering NPC interactions.
184
185        Args:
186            event: The pygame event to process.
187
188        Returns:
189            Updated scene with only player event processing.
190        """
191        return replace(self, player=self.player.event(event))
192
193    def tick(self, time_delta: Millisecond) -> Scene:
194        """Update the scene state over time.
195
196        This method updates the scene and handles ongoing events.
197        If there's a next event to process, it will be executed.
198
199        Args:
200            time_delta: Time elapsed since last update in milliseconds.
201
202        Returns:
203            Updated scene state.
204        """
205        if self._next_event:
206            return self._next_event
207        return self.tick_without_event(time_delta)
208
209    def tick_without_event(self, time_delta: Millisecond) -> Self:
210        """Update scene state without processing events.
211
212        Args:
213            time_delta: Time elapsed since last update in milliseconds.
214
215        Returns:
216            Updated scene with player and NPCs updated.
217        """
218        return replace(
219            self,
220            player=self.player.tick(time_delta),
221            npcs=tuple(n.tick(time_delta) for n in self.npcs),
222        )
223
224    def send(self, event: RpgEventGenerator, result: Any = None) -> Self:
225        """Continue event execution and optionally send the result of the current event.
226
227        This method is used to continue the execution of an event generator
228        and can pass a result value to the generator.
229
230        Args:
231            event: The generator to generate the next event.
232            result: Result of the current event, passing to the generator.
233
234        Returns:
235            Scene that shall continue with the next event.
236        """
237        return replace(self, _event_generator=event, _event_result=result)
238
239    @cached_property
240    def _collided_npc(self) -> NpcOnScreen | None:
241        """Find the NPC that the player is currently colliding with.
242
243        Returns:
244            The NPC the player is colliding with, or None if no collision.
245        """
246        for npc in self.npcs:
247            if self._collide(npc):
248                return npc
249        return None
250
251    def _collide(self, npc: NpcOnScreen) -> bool:
252        """Check if the player is colliding with an NPC.
253
254        Args:
255            npc: The NPC to check collision with.
256
257        Returns:
258            True if the player is colliding with the NPC, False otherwise.
259        """
260        return npc.draw_on_screen.rectangle.collide(
261            self.player.draw_on_screen.rectangle
262        )
263
264    def _trigger(self, npc: NpcOnScreen) -> Self:
265        """Trigger an event with the specified NPC.
266
267        This method prepares the scene for an NPC interaction by
268        setting up the NPC and player in event mode.
269
270        Args:
271            npc: The NPC to trigger an event with.
272
273        Returns:
274            Updated scene with the NPC event triggered.
275        """
276        triggered_npc = npc.start_event(self.player)
277        npcs = tuple(
278            triggered_npc if n.spec.object_name == npc.spec.object_name else n
279            for n in self.npcs
280        )
281        return replace(
282            self,
283            player=self.player.start_event(triggered_npc),
284            npc=triggered_npc,
285            npcs=npcs,
286        )
287
288    @cached_property
289    def _next_event(self) -> Scene | None:
290        """Get the scene for the next event.
291
292        This method continues the execution of the current event generator
293        and returns the next scene in the event sequence.
294
295        Returns:
296            The next scene to continue event execution, or None if no event.
297        """
298        if not self._event_generator:
299            return None
300
301        try:
302            create_next_scene = self._event_generator.send(self._event_result)
303            return create_next_scene(self._event_generator, self)
304        except StopIteration:
305            return self._clear_event
306
307    @cached_property
308    def _clear_event(self) -> Self:
309        """Clear the current event and return to normal scene state.
310
311        This method is called when an event sequence completes and
312        returns the scene to its normal state.
313
314        Returns:
315            Scene with event state cleared.
316        """
317        logger.debug(t"Event {self._event_generator.__name__} completed.")
318        return replace(
319            self,
320            player=self.player.complete_event,
321            npcs=self._completed_npcs,
322            npc=None,
323            _event_generator=None,
324            _event_result=None,
325        )
326
327    @cached_property
328    def _completed_npcs(self) -> tuple[NpcOnScreen, ...]:
329        """Get NPCs with their events completed.
330
331        Returns:
332            Tuple of NPCs with their event states completed.
333        """
334        return tuple(n.complete_event for n in self.npcs)

362@export
363@dataclass_with_instance_init
364class NpcSpec(CharacterSpec):
365    """Base class to define NPC specifications.
366
367    This class defines the complete specification for an NPC, including
368    its appearance, movement behavior, and event interactions.
369
370    Attributes:
371        event: Event specification for player/NPC interactions.
372        move_speed: Movement speed in pixels per millisecond.
373        idle_duration: Duration to stay idle in milliseconds.
374        move_duration: Duration to move in milliseconds.
375        cyclic_walk: Whether the NPC should walk in cycles.
376
377    Example:
378        ```python
379        npc_spec = NpcSpec(
380            name="Merchant",
381            drawing=merchant_drawing,
382            event=shop_dialog,
383            move_speed=0.05,
384            idle_duration=2000,
385            move_duration=1000,
386            cyclic_walk=True
387        )
388        ```
389    """
390
391    event: RpgEventSpec
392    move_speed: PixelPerMillisecond = field(
393        default_factory=lambda: config().character.move_speed
394    )
395    idle_duration: Millisecond = field(
396        default_factory=lambda: config().character.idle_duration
397    )
398    move_duration: Millisecond = field(
399        default_factory=lambda: config().character.move_duration
400    )
401    cyclic_walk: bool = True
402
403    @cached_property
404    def generator(self) -> Callable[[*RpgEventSpecParams], RpgEventGenerator]:
405        """Create a generator function from the event specification.
406
407        This method transforms the event specification into a generator
408        function that can be used for event execution.
409
410        Returns:
411            A callable that creates an event generator from the specification.
412        """
413
414        def yield_event(*args: Any, **kwargs: Any) -> RpgEventGenerator:
415            fun = self.event
416            ctx = fun.__globals__ | {
417                v: c.cell_contents
418                for v, c in zip(fun.__code__.co_freevars, fun.__closure__ or ())
419            }
420            exec(transform_and_compile(fun), ctx)
421            return ctx[fun.__name__](*args, **kwargs)
422
423        return yield_event
