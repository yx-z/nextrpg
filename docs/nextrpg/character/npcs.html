 51class MovingNpcOnScreen(NpcOnScreen, MovingCharacterOnScreen):
 52    """
 53    Moving NPC interface.
 54
 55    Arguments:
 56        `path`: Polygon representing the path of the NPC.
 57
 58        `idle_duration`: Duration of the idle state.
 59
 60        `move_duration`: Duration of the moving state.
 61    """
 62
 63    path: Polygon
 64    idle_duration: Millisecond
 65    move_duration: Millisecond
 66    _idle_timer: Timer = instance_init(lambda self: Timer(self.idle_duration))
 67    _move_timer: Timer = instance_init(lambda self: Timer(self.move_duration))
 68    _is_moving: bool = False
 69
 70    @override
 71    def tick(self, time_delta: Millisecond) -> Self:
 72        if self._is_moving:
 73            move_timer = self._move_timer.tick(time_delta)
 74            idle_timer = self._idle_timer
 75        else:
 76            move_timer = self._move_timer
 77            idle_timer = self._idle_timer.tick(time_delta)
 78
 79        is_moving = self._is_moving
 80        if self._is_moving and move_timer.expired:
 81            is_moving = False
 82        if not self._is_moving and idle_timer.expired:
 83            is_moving = True
 84
 85        return replace(
 86            super().tick(time_delta),
 87            _idle_timer=idle_timer.reset() if is_moving else idle_timer,
 88            _move_timer=move_timer.reset() if not is_moving else move_timer,
 89            _is_moving=is_moving,
 90        )
 91
 92    @cached_property
 93    @override
 94    def is_moving(self) -> bool:
 95        return self._is_moving

 97    @override
 98    def move(self, time_delta: Millisecond) -> Coordinate:
 99        return self.coordinate

124    def trigger(self, player: PlayerOnScreen, scene: Scene) -> Scene | None:
125        """
126        Trigger the NPC event.
127
128        `Arguments`:
129            `player`: The current player.
130
131            `scene`: The current scene.
132
133        Returns:
134            `Scene`: The scene after the event is triggered, if any.
135        """
136        if npc := self._collided_npc(player):
137            logger.debug(t"Collied with {npc.name}")
138            generator = _wrap_spec(npc.event_spec)(player, npc, self)
139            return next(generator)(generator, scene)
140        return None

142    def tick(self, time_delta: Millisecond) -> Self:
143        """
144        Update the npc state for a single game loop.
145
146        Arguments:
147            `time_delta`: The elapsed time since the last update.
148
149        Returns:
150            `Npcs`: The updated npc state after the step.
151        """
152        return replace(
153            self,
154            list=[n.tick(time_delta) for n in self.list],
155        )

157    def _init(self, spec: NpcSpec) -> NpcOnScreen:
158        obj = self.map_helper.get_object(spec.name)
159        return NpcOnScreen(
160            character=spec.drawing,
161            coordinate=Coordinate(obj.x, obj.y),
162            event_spec=spec.event_spec,
163            name=spec.name,

166    def _init_move(self, spec: MovingNpcSpec) -> MovingNpcOnScreen:
167        obj = self.map_helper.get_object(spec.name)
168        collisions = (
169            self.map_helper.collisions if spec.observe_collisions else []
170        )
171        return MovingNpcOnScreen(
172            character=spec.drawing,
173            coordinate=Coordinate(obj.x, obj.y),
174            collisions=collisions,
175            move_speed=spec.move_speed,
176            name=spec.name,
177            event_spec=spec.event_spec,
178            path=get_polygon(obj),
179            idle_duration=spec.idle_duration,
180            move_duration=spec.move_duration,
181        )

183    def _collided_npc(self, player: PlayerOnScreen) -> NpcOnScreen | None:
184        collide = (n for n in self.list if _collide_with_player(n, player))
185        return next(collide, None)

241def _wrap_spec(
242    fun: RpgEventSpec,
243) -> Callable[[PlayerOnScreen, NpcOnScreen, Npcs], RpgEventGenerator]:
244    @wraps(fun)
245    def wraped(
246        player: PlayerOnScreen, npc: NpcOnScreen, npcs: Npcs
247    ) -> RpgEventGenerator:
248        src = getsource(fun)
249        tree = fix_missing_locations(_yield.visit(parse(src)))
250        code = compile(tree, "<npcs>", "exec")
251        ctx = fun.__globals__
252        exec(code, ctx)
253        return ctx[fun.__name__](player, npc, npcs)
254
255    return wraped
