 22class CharacterOnScreen(ABC):
 23    """
 24    Represents a character that can be displayed and moved on screen.
 25
 26    Handles character movement, collision detection, and event processing for
 27    character interactions with the game environment.
 28
 29    Arguments:
 30        `character_drawing`: The visual representation of the character.
 31
 32        `coordinate`: The current position of the character on screen.
 33
 34        `speed`: Movement speed of the character in pixels.
 35
 36        `collisions`: List of polygons representing collision boundaries.
 37    """
 38
 39    character: CharacterDrawing
 40    coordinate: Coordinate
 41
 42    @cached_property
 43    def draw_on_screen(self) -> DrawOnScreen:
 44        """
 45        Creates drawable representations of the character and visuals.
 46
 47        Returns:
 48            `CharacterAndVisuals`: A tuple containing the character's drawable
 49            representation and any associated visual elements.
 50        """
 51        return DrawOnScreen(self.coordinate, self.character.drawing)
 52
 53    @abstractmethod
 54    def tick(self, time_delta: Millisecond) -> Self:
 55        """
 56        Update the character's state for a single game loop.
 57
 58        Arguments:
 59            `time_delta`: The elapsed time since the last update.
 60
 61        Returns:
 62            `CharacterOnScreen`: The updated character state after the step.
 63        """
 64
 65    def trigger(self, character: CharacterOnScreen) -> Self:
 66        direction = character.coordinate.relative_to(self.coordinate)
 67        return replace(self, character=self.character.turn(direction))

 99    @override
100    def tick(self, time_delta: Millisecond) -> Self:
101        """
102        Update the character's state for a single game step/frame.
103
104        Calculates movement based on currently pressed keys, handles collision
105        detection, and updates the character's drawing state (moving or idle).
106
107        Arguments:
108            `time_delta`: The time that has passed since
109                the last update, used for calculating movement distance.
110
111        Returns:
112            `CharacterOnScreen`: The updated character state after the step.
113        """
114        moved_coord = self.move(time_delta) if self.moving else None
115        character = (
116            self.character.move(time_delta)
117            if self.moving
118            else self.character.idle(time_delta)
119        )
120        coordinate = (
121            moved_coord
122            if moved_coord and self._can_move(moved_coord)
123            else self.coordinate
124        )
125        return replace(self, character=character, coordinate=coordinate)
126
127    def _can_move(self, coordinate: Coordinate) -> bool:
128        if (debug := config().debug) and debug.ignore_map_collisions:
129            return True
130
131        rect = DrawOnScreen(coordinate, self.character.drawing).rectangle
132        hit_coords = {
133            Direction.LEFT: {rect.bottom_left, rect.center_left},
134            Direction.RIGHT: {rect.bottom_right, rect.center_right},
135            Direction.DOWN: {
136                rect.bottom_right,
137                rect.bottom_center,
138                rect.bottom_left,
139            },
140            Direction.UP: {rect.center_right, rect.center, rect.center_left},
141            Direction.UP_LEFT: {rect.center_left},
142            Direction.UP_RIGHT: {rect.center_right},
143            Direction.DOWN_LEFT: {
144                rect.bottom_left,
145                rect.center_left,
146                rect.bottom_center,
147            },
148            Direction.DOWN_RIGHT: {
149                rect.bottom_right,
150                rect.center_right,
151                rect.bottom_center,
152            },
153        }[self.character.direction]
154
155        if collision_and_coord := self._collide(hit_coords):
156            collision, coord = collision_and_coord
157            logger.debug(
158                t"Collision {coord} and {collision.points}",
159                duration=FROM_CONFIG,
160            )
161            return False
162        return True
163
164    def _collide(
165        self, hit_coords: set[Coordinate]
166    ) -> _CollisionAndCoord | None:
167        collided = (
168            _CollisionAndCoord(collision, coord)
169            for collision, coord in product(self.collisions, hit_coords)
170            if collision.contain(coord)
171        )
172        return next(collided, None)
